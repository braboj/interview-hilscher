105 {
106  char input;

108  vector data;

110

112  input = '\0';
113  state = Entering;
114  vector_init(&data);
115  printf("\n%-8s %s",FSM_B_STATE_TEXT[Entering],FSM_PROMPT);
116 
117  //FSM switching
118  while (1)
119  {
120  input = getch();
121 
122  s = sizeof(FSM)/sizeof(FSM_TABLE);
123 
124  //Loop through table entries
125  for(int i=0;i<s;i++)
126  {
127  //Current state found in the table
128  if (FSM[i].state==state)
129  {
130  //Event found in the table
131  if (FSM[i].event==input)
132  {
133  FSM[i].output(&data, &input, state);
134  state = FSM[i].next;
135  break;
136  }
137 
138  //Event not found in the table
139  if (FSM[i].event==OtherEvent)
140  {
141  FSM[i].output(&data, &input, state);
142  break;
143  }
144 
145  }
146  }
147  }
148 }
